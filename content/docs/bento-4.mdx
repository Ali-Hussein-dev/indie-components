## Variant 1

<Tabs items={["Preview", "React"]}>
  <Tab value="Preview">
    <Bento_4_v1 />
  </Tab>
  <Tab value="React">

```jsx
import { cn } from "@/src/utils/cn"

export const Cell = ({ i = 0 }) => {
  return (
    <div className="size-full rounded-lg center bg-zinc-200 dark:bg-zinc-900 text-xl">
      {i}
    </div>
  )
}
const cells = [1, 2, 3, 4]
//======================================
export const Bento_4_v1 = () => {
  return (
    <div>
      <div className="grid md:grid-cols-4 gap-2">
        {cells.map((n, i) => (
          <div
            key={n}
            className={cn(
              "p-1 rounded-lg h-32",
              i == 0 && "md:col-span-3",
              i == 2 && "md:col-start-1",
              i == 3 && "md:col-start-2 md:col-span-4"
            )}
          >
            <Cell i={i + 1} />
          </div>
        ))}
      </div>
    </div>
  )
}
```
